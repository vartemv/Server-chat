38 UDPhandler(int ret, int t, sockaddr_in client, int kill) {
39 this->retransmissions = ret;
40 this->timeout_chat = t;
41 this->global_counter = 0;
42 this->client_socket = socket(AF_INET, SOCK_DGRAM, 0);
43 if (this->client_socket < 0) {
44 perror("Problem with creating response socket");
45 exit(EXIT_FAILURE);
46 }
47
48
49 epoll_fd = epoll_create1(0);
50 if (epoll_fd == -1) {
51 std::cerr << "Failed to create epoll file descriptor\n";
52 exit(EXIT_FAILURE);
53 }
54
55 // setup epoll event
56 struct epoll_event ev;
57 ev.events = EPOLLIN | EPOLLET;
58 ev.data.fd = this->client_socket;
59
60 // add socket file descriptor to epoll
61 if (epoll_ctl(epoll_fd, EPOLL_CTL_ADD, this->client_socket, &ev) == -1) {
62 std::cerr << "Failed to add file descriptor to epoll\n";
63 close(epoll_fd);
64 exit(EXIT_FAILURE);
65 }
66
67 ev.data.fd = kill;
68 if (epoll_ctl(epoll_fd, EPOLL_CTL_ADD, kill, &ev) < 0) {
69 std::cerr << "Unable to add socket to epoll\n";
70 exit(EXIT_FAILURE);
71 }
72
73 auth = false;
74
75 client_addr = client;
76
77 channel_name = "general";
78
79 }
