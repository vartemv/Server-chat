6void UDPhandler::handleUDP(uint8_t *buf, sockaddr_in client_addr, int length, int retransmissions, int timeout,
7 int *busy, std::stack<UserInfo> *s, synch *synch_var, int signal_listener) {
8
9 UDPhandler udp(retransmissions, timeout, client_addr, signal_listener);
10
11 bool end = false;
12
13 std::thread sender(read_queue, s, &end, synch_var, busy, &udp);
14
15 uint8_t internal_buf[2048];
16 logger(udp.client_addr, "AUTH", "RECV");
17 udp.send_confirm(buf);
18 int result = udp.respond_to_auth(buf, length, s, synch_var);
19
20 if (result != -1) {
21 while (true) {
22 int length_internal = udp.wait_for_the_incoming_connection(internal_buf);
23 if (length_internal == -1) {
24 uint8_t buf_int[256];
25 int length_int = udp.create_bye(buf_int);
26 udp.send_message(buf_int, length_int, true);
27 break;
28 }
29 if (!udp.decipher_the_message(internal_buf, length_internal, s, synch_var)) {
30 break;
31 }
32 }
33 }
34
35 if (synch_var->usernames.find(udp.user_n) != synch_var->usernames.end())
36 synch_var->usernames.erase(udp.user_n);
37
38 end = true;
39 {
40 std::lock_guard<std::mutex> lock(synch_var->mtx);
41 synch_var->ready = true;
42 }
43 synch_var->cv.notify_all();
44
45 sender.join();
46 close(udp.client_socket);
47}

49void read_queue(std::stack<UserInfo> *s, bool *terminate, synch *synch_vars, int *busy, UDPhandler *udp) {
50 while (!*terminate) {
51 std::unique_lock<std::mutex> lock(synch_vars->mtx);
52 synch_vars->cv.wait(lock, [&synch_vars] { return synch_vars->ready; });
53
54 synch_vars->waiting.lock();
55 synch_vars->finished++;
56 synch_vars->waiting.unlock();
57
58 if (!s->empty() && udp->auth) {
59
60 synch_vars->waiting.lock();
61 UserInfo new_uf = s->top();
62 synch_vars->waiting.unlock();
63
64 if (new_uf.tcp) {
65 if (new_uf.channel == udp->channel_name) {
66 uint8_t buf[3048];
67 int length = udp->convert_from_tcp(buf, new_uf.buf);
68 udp->send_message(buf, length, false);
69 }
70 } else {
71 if ((new_uf.client.sin_port != udp->client_addr.sin_port && new_uf.channel == udp->channel_name)) {
72 udp->send_message(new_uf.buf, new_uf.length, false);
73 }
74 }
75 }
76 if (synch_vars->finished == *busy) {
77 synch_vars->finished = 0;
78 synch_vars->ready = false;
79 if (!s->empty())
80 s->pop();
81 }
82 lock.unlock();
83 std::this_thread::sleep_for(std::chrono::milliseconds(100));
84 lock.lock();
85
86 }

89bool UDPhandler::decipher_the_message(uint8_t *buf, int length, std::stack<UserInfo> *s, synch *synch_var) {
90 if (!this->auth) {
91 if (buf[0] != 0x02) {
92 if (buf[0] != 0xFF) {
93 if (buf[0] != 0xFE) {
94 uint8_t buf_int[1024];
95 std::string message = "You should log-in before doing anything else";
96 std::string name = "Server";
97 int length = this->create_message(buf_int, message, false, name);
98 this->send_message(buf_int, length, false);
99 return true;
100 }
101 }
102 }
103 }
104
105 switch (buf[0]) {
106 case 0x00://CONFIRM
107 break;
108 case 0x02://AUTH
109 logger(this->client_addr, "AUTH", "RECV");
110 send_confirm(buf);
111 if (!this->auth) {
112 respond_to_auth(buf, length, s, synch_var);
113 } else {
114 uint8_t buf_err[1024];
115 std::string message = "Already authed";
116 std::string name = "Server";
117 int length_err = this->create_message(buf_err, message, true, name);
118 this->send_message(buf_err, length_err, false);
119 }
120 break;
121 case 0x03://JOIN
122 logger(this->client_addr, "JOIN", "RECV");
123 send_confirm(buf);
124 respond_to_join(buf, length, s, synch_var);
125 break;
126 case 0x04://MSG
127 logger(this->client_addr, "MSG", "RECV");
128 send_confirm(buf);
129 this->message(buf, length, s, synch_var, this->channel_name);
130 break;
131 case 0xFF://BYE
132 if (this->auth) {
133 this->client_leaving(s, synch_var);
134 }
135 logger(this->client_addr, "BYE", "RECV");
136 send_confirm(buf);
137 return false;
138 case 0xFE://ERR
139 if (this->auth) {
140 this->client_leaving(s, synch_var);
141 }
142 logger(this->client_addr, "ERR", "RECV");
143 send_confirm(buf);
144 return false;
145 default:
146 uint8_t buf[1024];
147 std::string message = "Unknown instruction";
148 std::string name = "Server";
149 int length_err = this->create_message(buf, message, true, name);
150 this->send_message(buf, length_err, false);
151 return false;
152 }
153 return true;

336int UDPhandler::wait_for_the_incoming_connection(uint8_t *buf_out, int timeout) {
337 int event_count = epoll_wait(this->epoll_fd, this->events, 2, timeout);
338
339 if (event_count == -1) {
340 perror("epoll_wait");
341 close(this->epoll_fd);
342 exit(EXIT_FAILURE);
343 } else if (event_count > 0) {
344 socklen_t len_client = sizeof(this->client_addr);
345 for (int j = 0; j < event_count; j++) {
346 if (events[j].data.fd == this->client_socket) { // check if EPOLLIN event has occurred
347 int n = recvfrom(this->client_socket, buf_out, 1024, 0, (struct sockaddr *) &this->client_addr,
348 &len_client);
349 if (n == -1) {
350 std::cerr << "recvfrom failed. errno: " << errno << '\n';
351 continue;
352 }
353 if (n > 0) {
354 return n;
355 }
356 } else {
357 return -1;
358 }
359 }
360 }
361 return 0;

388bool UDPhandler::buffer_validation(uint8_t *buf, int message_length, int start_position, int minimal_length,
389 int amount_of_fields, int first_limit, int second_limit, int third_limit) {
390
391 if (message_length < minimal_length)
392 return false;
393
394 size_t i = start_position;
395
396 size_t count = 0;
397 while (i < message_length && buf[i] != 0x00 && count < first_limit) {
398 i++;
399 count++;
400 }
401
402 if (i >= message_length || buf[i] != 0x00 || count < 1) {
403 return false;
404 }
405 ++i;
406
407 count = 0;
408 while (i < message_length && buf[i] != 0x00 && count < second_limit) {
409 ++i;
410 ++count;
411 }
412
413 if (i >= message_length || buf[i] != 0x00 || count < 1) {
414 return false;
415 }
416
417 ++i;
418
419 if (amount_of_fields == 3) {
420 count = 0;
421 while (i < message_length && buf[i] != 0x00 && count < third_limit) {
422 ++i;
423 ++count;
424 }
425
426 if (i >= message_length || buf[i] != 0x00 || count < third_limit) {
427 return false;
428 }
429 }
430
431 return true;
432}

459

463 message.push_back(static_cast<char>(tcp_buf[i]));
