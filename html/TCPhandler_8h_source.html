22 TCPhandler(int s, sockaddr_in c, int kill) {
23 this->channel_name = "general";
24 this->client_socket = s;
25
26 epoll_fd = epoll_create1(0);
27 if (epoll_fd == -1) {
28 std::cerr << "Failed to create epoll file descriptor\n";
29 exit(EXIT_FAILURE);
30 }
31
32 // setup epoll event
33 struct epoll_event ev;
34 ev.events = EPOLLIN;
35 ev.data.fd = this->client_socket;
36
37 // add socket file descriptor to epoll
38 if (epoll_ctl(epoll_fd, EPOLL_CTL_ADD, this->client_socket, &ev) == -1) {
39 std::cerr << "Failed to add file descriptor to epoll\n";
40 close(epoll_fd);
41 exit(EXIT_FAILURE);
42 }
43
44 ev.data.fd = kill;
45 if (epoll_ctl(epoll_fd, EPOLL_CTL_ADD, kill, &ev) < 0) {
46 std::cerr << "Unable to add socket to epoll\n";
47 exit(EXIT_FAILURE);
48 }
49
50 client_addr = c;
51
52 auth = false;
53 }
